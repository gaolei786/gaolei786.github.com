区间估计函数 


set.seed(123)
norm.conf.interval <- function(i)
{
  y <- rnorm(16, 5, 1)
  n <- length(y)
  z=qt(0.975, n - 1)
  u <- mean(y)
  s <- sd(y)
  ci <- c(u - s * z / n^0.5, u + s * z / n^0.5, (u-u0)/(s/n^0.5))
}
ci <- sapply(1:200, norm.conf.interval)
